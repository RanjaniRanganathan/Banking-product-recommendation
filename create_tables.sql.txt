CREATE TABLE CUSTOMERS
(
CUSTOMER_ID NUMBER PRIMARY KEY,
CUSTOMER_NAME VARCHAR2(50),
CUSTOMER_AGE NUMBER,
INCOME NUMBER,
RISK_TOLERANCE VARCHAR2(20)CHECK (RISK_TOLERANCE IN ('Low', 'Medium', 'High')),
INVESTMENT_GOAL VARCHAR2(50)
);


CREATE TABLE INVESTMENT_PRODUCTS 
(
    PRODUCT_ID NUMBER PRIMARY KEY,
    PRODUCT_NAME VARCHAR2(50),
    RISK_LEVEL VARCHAR2(20) CHECK (RISK_LEVEL IN ('Low', 'Medium', 'High')),
    MIN_DURATION NUMBER,
    RETURN_RATE NUMBER
);


CREATE TABLE INVESTMENT_RECOMMENDATIONS
 (
    RECOMMENDATION_ID NUMBER PRIMARY KEY,
    CUSTOMER_ID NUMBER,
    PRODUCT_ID NUMBER,
    RECOMMENDED_BY VARCHAR2(100),
    RECOMMENDATION_DATE DATE,
    FOREIGN KEY (CUSTOMER_ID) REFERENCES CUSTOMERS(CUSTOMER_ID),
    FOREIGN KEY (PRODUCT_ID) REFERENCES INVESTMENT_PRODUCTS(PRODUCT_ID)
);
